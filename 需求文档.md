心上人 - 微信小程序产品需求文档（PRD）
一、产品概述
1.1 产品定位
"心上人"是一款专注于情侣关系维护与情感升温的微信小程序。通过AI智能生成情书、生活点滴记录、纪念日提醒、本地情侣活动推荐等功能，帮助用户追求心仪对象或维系现有感情。
1.2 目标用户
追求阶段：希望打动心仪对象的单身用户
热恋阶段：希望保持新鲜感、创造共同回忆的情侣
稳定阶段：希望维系感情、避免平淡期的伴侣
1.3 核心价值主张
"让每一天都是情人节" —— 通过AI赋能的情感表达与智能提醒，降低用户维护感情的门槛，提升情感表达质量。

3.1.1 用户登录/注册
实现逻辑：
plain
复制
1. 前端调用 wx.login() 获取 code
2. 将 code 发送到云函数 login
3. 云函数用 code 调用微信 auth.code2Session 接口获取 openid + session_key
4. 查询 MySQL users 表：
    - 存在：更新 session_key，返回 user_id + token（JWT）
    - 不存在：插入新用户，返回 user_id + token
5. 前端存储 token 到 Storage，后续请求携带 token
6. 调用 wx.getUserProfile() 获取昵称头像（需用户授权），更新用户信息
   云函数代码逻辑（伪代码）：
   JavaScript
   复制
   // 云函数 login
   const jwt = require('jsonwebtoken');
   const mysql = require('mysql2/promise');

exports.main = async (event, context) => {
const { code, userInfo } = event;

// 1. 获取 openid
const wxRes = await request(`https://api.weixin.qq.com/sns/jscode2session?appid=${APPID}&secret=${SECRET}&js_code=${code}&grant_type=authorization_code`);
const { openid, session_key, unionid } = wxRes;

// 2. 连接数据库
const db = await mysql.createConnection(DB_CONFIG);

// 3. 查询或创建用户
let [rows] = await db.execute('SELECT * FROM users WHERE openid = ?', [openid]);
let userId;

if (rows.length === 0) {
// 新用户注册
const [result] = await db.execute(
'INSERT INTO users (openid, unionid, nickname, avatar_url, gender) VALUES (?, ?, ?, ?, ?)',
[openid, unionid, userInfo?.nickName, userInfo?.avatarUrl, userInfo?.gender]
);
userId = result.insertId;
} else {
// 更新 session_key
userId = rows[0].id;
await db.execute('UPDATE users SET session_key = ? WHERE id = ?', [session_key, userId]);
}

// 4. 生成 JWT Token
const token = jwt.sign({ userId, openid }, JWT_SECRET, { expiresIn: '7d' });

await db.end();
return { code: 0, data: { token, userId, isNew: rows.length === 0 } };
};

3.4.3 提醒触发机制
定时任务设计（云函数 + 微信云托管定时触发器）：
JavaScript
复制
// 云函数 dailyReminderTrigger，配置 cron: 0 5 0 * * * （每天00:05）
exports.main = async (event, context) => {
const now = new Date();
const today = formatDate(now); // YYYY-MM-DD
const currentTime = formatTime(now); // HH:mm:ss

const db = await mysql.createConnection(DB_CONFIG);

// 1. 查询今日到期的提醒（提前天数已考虑）
const [reminders] = await db.execute(`
    SELECT r.*, u.openid as user_openid, p.openid as partner_openid
    FROM reminders r
    JOIN users u ON r.user_id = u.id
    LEFT JOIN users p ON r.partner_id = p.id
    WHERE r.status = 1 
      AND DATE(r.target_date) <= DATE_ADD(?, INTERVAL r.advance_days DAY)
      AND r.notify_time <= ?
      AND (r.last_notify_date IS NULL OR r.last_notify_date < ?)
  `, [today, currentTime, today]);

// 2. 批量发送微信订阅消息
for (const reminder of reminders) {
const templateData = {
thing1: { value: reminder.title },           // 提醒事项
time2: { value: reminder.target_date },      // 提醒时间
thing3: { value: `还有${reminder.advance_days}天，别忘了准备惊喜哦~` }  // 备注
};

    // 通知用户自己
    if (reminder.notify_target & 1) {
      await sendSubscribeMessage(reminder.user_openid, REMINDER_TEMPLATE_ID, templateData);
    }
    // 通知对象
    if (reminder.partner_openid && (reminder.notify_target & 2)) {
      await sendSubscribeMessage(reminder.partner_openid, REMINDER_TEMPLATE_ID, templateData);
    }
    
    // 更新最后通知日期，避免重复发送
    await db.execute('UPDATE reminders SET last_notify_date = ? WHERE id = ?', [today, reminder.id]);
}

await db.end();
return { processed: reminders.length };
};

// utils/kimiApi.js
const axios = require('axios');

class KimiService {
constructor() {
this.apiKey = process.env.KIMI_API_KEY;
this.baseURL = 'https://api.moonshot.cn/v1';
}

async generateLetter(prompt, temperature = 0.7) {
try {
const response = await axios.post(`${this.baseURL}/chat/completions`, {
model: 'moonshot-v1-8k',  // 或 moonshot-v1-32k
messages: [
{ role: 'system', content: '你是一位专业的情感文案师，擅长撰写真挚动人的情书和情侣建议。' },
{ role: 'user', content: prompt }
],
temperature,
max_tokens: 800
}, {
headers: {
'Authorization': `Bearer ${this.apiKey}`,
'Content-Type': 'application/json'
}
});

      return response.data.choices[0].message.content;
    } catch (error) {
      console.error('Kimi API Error:', error.response?.data || error.message);
      throw new Error('AI生成失败，请稍后重试');
    }
}

async generateRecommendations(prompt) {
const content = await this.generateLetter(prompt, 0.5);
// 解析JSON，如失败则使用正则提取
try {
return JSON.parse(content);
} catch (e) {
return this.parseRecommendationsFallback(content);
}
}

// 正则提取备用方案
const foodMatch = content.match(/美食[：:](.+)/);
const activityMatch = content.match(/活动[：:](.+)/);
return {
food: { name: foodMatch?.[1] || '本地特色菜', restaurant: '推荐餐厅', reason: '适合情侣' },
activity: { name: activityMatch?.[1] || '看电影', location: '市中心', reason: '浪漫约会' },
todos: ['一起做饭', '散步聊天']
};
}
}

async function sendSubscribeMessage(openid, templateId, data, page = '') {
// 1. 获取接口调用凭证 access_token
const tokenRes = await axios.get(`https://api.weixin.qq.com/cgi-bin/token?grant_type=client_credential&appid=${APPID}&secret=${APPSECRET}`);

touser: openid,
template_id: templateId,
page: page,
data: data,
miniprogram_state: 'formal', // formal/trial/developer
lang: 'zh_CN'
});

if (sendRes.data.errcode !== 0) {
// 记录失败日志，如用户未授权则减少订阅次数
console.error('Send message failed:', sendRes.data);
return { success: false, errcode: sendRes.data.errcode };
}

return { success: true };
}
